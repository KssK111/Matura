FROM SPRZET;

W którym roku wyprodukowano najnowszy sprzęt dostępny w wypożyczalni? Ile jest sztuk 
narzędzi z tego roku?
SELECT ROK_PRODUKCJI, COUNT(ROK_PRODUKCJI) AS LICZBA_SZTUK
FROM SPRZET
ORDER BY ROK_PRODUKCJI DESC
LIMIT 1
